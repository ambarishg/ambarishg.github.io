Chapter 82 Sentiment analysis by Inspection Text

We examine the positive and the negative Text.We filtered out messages that had fewer than five words that contributed to sentiment.

sentiment_messages <- FoodInspectionsReduced %>%
  unnest_tokens(word, Violations) %>%
  filter(!word %in% stop_words$word) %>%
  inner_join(get_sentiments("afinn"), by = "word") %>%
  group_by(Results, InspectionID) %>%
  dplyr::summarize(sentiment = mean(score),
            words = n()) %>%
  ungroup() %>%
  filter(words >= 5)

FoodInspectionsReduced2 = FoodInspections %>%
